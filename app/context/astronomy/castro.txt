```python

# create a line plot of the density parallel to the x-axis that cuts through
the

# point of maximum density

_, c = ds.find_max("density") # find the location of the point of maximum
density ax = 0 # take cut parallel to x-axis

# use ortho_ray to cut through c

ray = ds.ortho_ray(ax, (c[1], c[2]))

# sort values by x so there are no discontinuities in the line plot

srt = np.argsort(ray['x'])

fig, ax = plt.subplots(figsize=(8,6)) ax.plot(ray['x'][srt],
ray['density'][srt]) ax.set_xlabel(r'$x$ (cm)') ax.set_ylabel(r'$\rho$
(g/cm$^3$)') ax.set_xlim([ray['x'][srt][0], ray['x'][srt][-1]])

fig.savefig('dens_plot.png') ```
